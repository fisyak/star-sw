TDataSet *CreateTable() {
  if (!gROOT->GetClass("St_beamSpot")) return 0;
  beamSpot_st row;
  St_beamSpot *tableSet = new St_beamSpot("beamSpot",1);
  memset(&row,0,tableSet->GetRowSize());
  row.X =     -0.344; row.sigma_X =      0.128;
  row.Y =     -2.024; row.sigma_Y =      0.073;
  row.Z =    200.490; row.sigma_Z =      0.296;
  tableSet->AddAt(&row);// PVxyz26p5GeV_fixedTarget_2021_P23id_calib.dEdx.root
  return (TDataSet *)tableSet;
}
