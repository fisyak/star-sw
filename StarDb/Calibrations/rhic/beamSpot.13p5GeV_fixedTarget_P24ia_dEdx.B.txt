TDataSet *CreateTable() {
  if (!gROOT->GetClass("St_beamSpot")) return 0;
  beamSpot_st row;
  St_beamSpot *tableSet = new St_beamSpot("beamSpot",1);
  memset(&row,0,tableSet->GetRowSize());
  row.X =     -0.321; row.sigma_X =      0.165;
  row.Y =     -2.056; row.sigma_Y =      0.078;
  row.Z =    200.128; row.sigma_Z =      0.381;
  tableSet->AddAt(&row);// PVxyz13p5GeV_fixedTarget_P24ia_dEdx.B.root
  return (TDataSet *)tableSet;
}
