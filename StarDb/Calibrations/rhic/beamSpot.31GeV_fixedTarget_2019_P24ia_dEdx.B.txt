TDataSet *CreateTable() {
  if (!gROOT->GetClass("St_beamSpot")) return 0;
  beamSpot_st row;
  St_beamSpot *tableSet = new St_beamSpot("beamSpot",1);
  memset(&row,0,tableSet->GetRowSize());
  row.X =     -0.366; row.sigma_X =      0.111;
  row.Y =     -2.010; row.sigma_Y =      0.073;
  row.Z =    200.033; row.sigma_Z =      0.278;
  tableSet->AddAt(&row);// PVxyz31GeV_fixedTarget_2019_P24ia_dEdx.B.root
  return (TDataSet *)tableSet;
}
