#include "tables/St_beamSpot_Table.h"
#endif
TDataSet *CreateTable() {
  if (!gROOT->GetClass("St_beamSpot")) return 0;
  beamSpot_st row;
  St_beamSpot *tableSet = new St_beamSpot("beamSpot",1);
  memset(&row,0,tableSet->GetRowSize());
  row.X =      0.001; row.sigma_X =      0.222;
  row.Y =     -1.984; row.sigma_Y =      0.089;
  row.Z =    200.533; row.sigma_Z =      0.278;
  tableSet->AddAt(&row);// PVxyz3p85GeV_fixedTarget.xrootd3.root
  return (TDataSet *)tableSet;
}
