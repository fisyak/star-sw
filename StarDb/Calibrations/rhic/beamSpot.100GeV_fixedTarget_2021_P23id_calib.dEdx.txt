TDataSet *CreateTable() {
  if (!gROOT->GetClass("St_beamSpot")) return 0;
  beamSpot_st row;
  St_beamSpot *tableSet = new St_beamSpot("beamSpot",1);
  memset(&row,0,tableSet->GetRowSize());
  row.X =     -0.517; row.sigma_X =      0.081;
  row.Y =     -1.993; row.sigma_Y =      0.070;
  row.Z =    200.516; row.sigma_Z =      0.297;
  tableSet->AddAt(&row);// PVxyz100GeV_fixedTarget_2021_P23id_calib.dEdx.root
  return (TDataSet *)tableSet;
}
