TDataSet *CreateTable() {

  beamSpot_st row;
  St_beamSpot *tableSet = new St_beamSpot("beamSpot",1);
  memset(&row,0,tableSet->GetRowSize());
  row.X =     -0.312; row.sigma_X =      0.186;
  row.Y =     -2.148; row.sigma_Y =      0.144;
  row.Z =    200.463; row.sigma_Z =      0.183;

}
