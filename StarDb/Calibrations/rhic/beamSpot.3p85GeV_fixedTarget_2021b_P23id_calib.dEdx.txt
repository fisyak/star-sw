TDataSet *CreateTable() {

  beamSpot_st row;
  St_beamSpot *tableSet = new St_beamSpot("beamSpot",1);
  memset(&row,0,tableSet->GetRowSize());
  row.X =     -0.490; row.sigma_X =      0.120;
  row.Y =     -2.018; row.sigma_Y =      0.076;
  row.Z =    200.033; row.sigma_Z =      0.278;
  tableSet->AddAt(&row);// PVxyz3p85GeV_fixedTarget_2021b_P23id_calib.dEdx.root

}
