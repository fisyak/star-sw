#endif
TDataSet *CreateTable() {
  if (!gROOT->GetClass("St_beamSpot")) return 0;
  beamSpot_st row;
  St_beamSpot *tableSet = new St_beamSpot("beamSpot",1);
  memset(&row,0,tableSet->GetRowSize());
  row.X =     -0.248; row.sigma_X =      0.138;
  row.Y =     -2.044; row.sigma_Y =      0.071;
  row.Z =    200.143; row.sigma_Z =      0.422;
  tableSet->AddAt(&row);// PVxyz31p2GeV_fixedTarget_P24ia_dEdx.root
  return (TDataSet *)tableSet;
}
